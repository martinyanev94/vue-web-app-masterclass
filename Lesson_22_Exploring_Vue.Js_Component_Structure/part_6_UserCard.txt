<template>
  <div class="user-card">
    <h2>{{ user.name }}</h2>
    <p>{{ user.email }}</p>
    <button @click="sendMessage">Send Message</button>
  </div>
</template>

<script lang="ts">
import { defineComponent } from 'vue';

export default defineComponent({
  props: {
    user: {
      type: Object,
      required: true,
    },
  },
  methods: {
    sendMessage() {
      this.$emit('message-sent', this.user.id);
    },
  },
});
</script>

<style scoped>
.user-card {
  border: 1px solid #ccc;
  padding: 10px;
  margin: 5px;
}
</style>

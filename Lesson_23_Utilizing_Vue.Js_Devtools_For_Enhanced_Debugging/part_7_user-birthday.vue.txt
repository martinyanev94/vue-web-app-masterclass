<template>
  <div>
    <h1>{{ user.name }}</h1>
    <p>{{ user.age }} years old</p>
    <p>Birthday in {{ birthdayYear }}!</p>
    <button @click="increaseAge">Happy Birthday!</button>
  </div>
</template>

<script lang="ts">
import { defineComponent, ref, computed } from 'vue';

export default defineComponent({
  setup() {
    const user = ref({
      name: 'Alice',
      age: 25,
    });

    const increaseAge = () => {
      user.value.age += 1;
    };

    const birthdayYear = computed(() => {
      const currentYear = new Date().getFullYear();
      return currentYear + (100 - user.value.age);
    });

    return { user, increaseAge, birthdayYear };
  },
});
</script>
